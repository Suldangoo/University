  useEffect( () => {
    const id = setInterval( () => {
      setDate( DateTime.now() );
    }, 500 );
    return () => clearInterval( id );
  }, [] );

  return (
    <Container source={image} resizeMode="stretch">
    <TimeText>{ date.toFormat( 'HH:mm' ) }</TimeText>
    <Second>{ date.toFormat( 'ss' ) }</Second>
    </Container>
  )
}
